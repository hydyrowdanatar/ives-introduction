export async function CityStateGet(Zipcode) {
    console.log ('CityStateGet Start - Zipcode:' + Zipcode);

    var CityStateResult = {
            "City": "",
            "State": ""
        }

    console.log('CSG Zipcode-' + Zipcode)
    /* await wixData.query("EstimatorInsuranceRates").eq("iSIndividual", "Yes").eq("zipcode", Zipcode).eq("deductibleAmount", 10000) */
    await wixData.query("EstimatorInsuranceRates").eq("zipcode", Zipcode)
    .find()
    .then(results => {
        console.log('CSG Query Result-' + results.items.length )
        if (results.items.length > 0) {
            CityStateResult = {
                "City": results.items[0].city,
                "State": results.items[0].state
            }
        }
    })
    .catch (error => console.log('Error-' + error))

    console.log ('CityStateGet End');
    
    return CityStateResult;
}

export function formatDollarsOnly(CurrencyAmount) {
	let Formatter = new Intl.NumberFormat('en-US', {
  		style: 'currency',
  		currency: 'USD',
  		minimumFractionDigits: 0, // (this suffices for whole numbers, but will print 2500.10 as $2,500.1)
  		maximumFractionDigits: 0, // (causes 2500.99 to be printed as $2,501)
	});
	//console.log("FDO CurrencyAmount-"+ CurrencyAmount)
	return Formatter.format(CurrencyAmount)
}

export async function MYNDDatabaseFunction() {

    console.log ('MYNDDatabaseFunction Start');

    // Event to update values in the database
	wixData.query("EstimatorInsuranceRates")
	.find()
	.then( (QueryResults) => {
		let TotalCount = QueryResults.totalCount;
		let AllItems = QueryResults.items
        
        console.log ('MDBF Total Count-' + TotalCount);

        var RecordCounter = 0;
		AllItems.forEach((RateRecord) => {
            RecordCounter = RecordCounter + 1;
            let RecordRemainder = RecordCounter % 100;
            if (RecordRemainder == 0) {
                console.log ('MDBF Record Count-' + RecordCounter);
            }
			RateRecord.iSInstitutional = "No";
			RateRecord.iSIndividual = "Yes";
			wixData.update("EstimatorInsuranceRates", RateRecord);
		})
	})
    console.log ('MYNDDatabaseFunction End');
}

export async function ReplacementCostCalculation(BuildingType, PropertySqFt, Zipcode) {
    var InputParametersOK;
    var ReplacementCostStatus;

    const PropertySqFtMin = 700;
    const PropertySqFtMax = 5000;

    var GarageSqft;
    var GarageCostFactor;
    var HVACCostFactor;
    var FireplaceAmount;
    var LowReplacementCost;
    var HighReplacementCost;

    var ReplacementCostResult = {
            "ReplacementCostStatus": '',
            "LowReplacementCost": 0.0,
            "HighReplacementCost": 0.0
        }
    console.log ('ReplacementCostCalculation Start');

    InputParametersOK = 'Y'
    if ((BuildingType == '') || isNaN(BuildingType) || (BuildingType !== 1)) {
        ReplacementCostStatus = 'Building Type MUST Be - 1 Single Family Residence.';
        InputParametersOK = 'N';
    } else if ((PropertySqFt == '') || isNaN(PropertySqFt) || (PropertySqFt < PropertySqFtMin) || (PropertySqFt > PropertySqFtMax)) {
        ReplacementCostStatus = 'Property Square Footage is NOT Correct, Square Footage MUST Be Between ' + PropertySqFtMin + ' and ' + PropertySqFtMax + '.';
        InputParametersOK = 'N';
    } else if ((Zipcode == '') || isNaN(Zipcode)) {
        ReplacementCostStatus = 'Zipcode is NOT Correct.';
        InputParametersOK = 'N';
    }

    console.log('ReplacementCostCalculation BuildingType-' + BuildingType + ', PropertySqft-' + PropertySqFt + ', Zipcode-' + Zipcode + ', Input Parameter OK-'+ InputParametersOK)
    
    if (InputParametersOK == 'Y') {

        const ZipQueryResult = await wixData.query("EstimatorZipcodeCost").eq("propertyZipcode", Zipcode).find();

        console.log('ReplacementCostCalculation Zipcode Query Result-' + ZipQueryResult.items.length )
        if (ZipQueryResult.items.length > 0) {
            
            const ZipcodeRateResult = ZipQueryResult.items[0];
            const NumPropertySqft = Number(PropertySqFt);
            const SqftQueryResult = await wixData.query("EstimatorSqftCost")
                .le("lowPropertySqft", NumPropertySqft)
                .ge("highPropertySqft", NumPropertySqft)
                .find();
            
            console.log('ReplacementCostCalculation Sqft Query Result-' + SqftQueryResult.items.length )
            if (SqftQueryResult.items.length > 0) {
                
                const LowCostPerSqftRate = SqftQueryResult.items[0].lowCostPerSqft;
                const HighCostPerSqftRate = SqftQueryResult.items[0].highCostPerSqft;

                console.log('ReplacementCostCalculation Sqft Low Cost Per Sqft-' + LowCostPerSqftRate + ', High Cost Per Sqft-' + HighCostPerSqftRate)
                
                const DefaultsQueryResult = await wixData.query("EstimatorRCTDefaults").find();
                console.log('ReplacementCostCalculation Defaults Query Result-' + DefaultsQueryResult.items.length )
                if (DefaultsQueryResult.items.length > 0) {
                    GarageSqft = DefaultsQueryResult.items[0].garageSqft;
                    GarageCostFactor = DefaultsQueryResult.items[0].garageCostFactor;
                    HVACCostFactor = DefaultsQueryResult.items[0].hvacCostFactor;
                    FireplaceAmount = DefaultsQueryResult.items[0].fireplaceAmount;
                }
                else {
                    GarageSqft = 0;
                    GarageCostFactor = 0.0;
                    HVACCostFactor = 0.0;
                    FireplaceAmount = 0.0;
                }
                console.log('ReplacementCostCalculation Defaults Garage Sqft-' + GarageSqft + ', Garage Cost Factor-' + GarageCostFactor + ', HVACCostFactor-' + HVACCostFactor + ', FireplaceAmount-' + FireplaceAmount)

                const ReplacementCostExtras = (GarageSqft * GarageCostFactor) + (PropertySqFt * HVACCostFactor) + FireplaceAmount;
                console.log('ReplacementCostCalculation ReplacementCostExtras-' + ReplacementCostExtras)

                LowReplacementCost = ((PropertySqFt * LowCostPerSqftRate) + ReplacementCostExtras) * (1 + (ZipcodeRateResult.weightedFactor / 100));
                const CalcLowPriceSqft = LowReplacementCost / PropertySqFt;
                if (CalcLowPriceSqft < ZipcodeRateResult.minPriceSqFt) {
                    LowReplacementCost = PropertySqFt * ZipcodeRateResult.minPriceSqFt
                }
                
                HighReplacementCost = ((PropertySqFt * HighCostPerSqftRate) + ReplacementCostExtras) * (1 + (ZipcodeRateResult.weightedFactor / 100));
                const CalcHighPriceSqft = HighReplacementCost / PropertySqFt;
                if (CalcHighPriceSqft < ZipcodeRateResult.minPriceSqFt) {
                    HighReplacementCost = PropertySqFt * ZipcodeRateResult.minPriceSqFt
                }
                console.log('ReplacementCostCalculation Sqft LowReplacementCost-' + LowReplacementCost + ', HighReplacementCost-' + HighReplacementCost)

                ReplacementCostResult = {
                    "ReplacementCostStatus": '',
                    "LowReplacementCost": LowReplacementCost,
                    "HighReplacementCost": HighReplacementCost
                }
            } else {
                ReplacementCostResult = {
                    "ReplacementCostStatus": 'Square Footage Not Found in Replacement Cost Database.',
                    "LowReplacementCost": 0.0,
                    "HighReplacementCost": 0.0
                }
            }
        } else {
            ReplacementCostResult = {
                "ReplacementCostStatus": 'Zipcode Not Found in Replacement Cost Database.',
                "LowReplacementCost": 0.0,
                "HighReplacementCost": 0.0
            }
        }
    } else {
        ReplacementCostResult = {
            "ReplacementCostStatus": ReplacementCostStatus,
            "LowReplacementCost": 0.0,
            "HighReplacementCost": 0.0
        }
    }
    console.log ('ReplacementCostCalculation end');
    return ReplacementCostResult;
}
