export function BindRequestClear() {
	console.log("BindRequestClear Start  In Policy Is Bound-" + PolicyIsBound)
	if (PolicyIsBound == 'Yes') {
		$w("#BindMessageDisplay").text = "Changing Accepted Quote - Requires Re-Acceptance.";
		$w("#BindMessageDisplay").show();
		PolicyIsBound = 'No';
		$w("#buttonBindRequest").show();
		$w("#buttonEvidence").hide();
		$w("#BindUserEmail").text ="";
		$w("#lblBindUserEmail").hide();
		$w("#BindUserEmail").hide();
	}
	console.log("BindRequestClear  End Policy Is Bound-" + PolicyIsBound)
}

export function BindRequestLockUnlock() {
	console.log("BindRequestLockUnlock Start Bind Request Date-" + BindRequestorDate)
	if (BindRequestorDate == '' || BindRequestorDate == undefined) {
		console.log("BindRequestLockUnlock Zipcode Unlock");
		$w("#PropertyZipcodeInput").readOnly = false;
		$w("#ProposalStreetAddressInput").readOnly = false;
		$w("#InfoPropertyAddress").readOnly = false;
	} else {
		$w("#PropertyZipcodeInput").readOnly = true;
		$w("#ProposalStreetAddressInput").readOnly = true;
		$w("#InfoPropertyAddress").readOnly = true;
		console.log("BindRequestLockUnlock Zipcode Lock");
	}
	console.log("BindRequestLockUnlock End");
}

export function formatDollarsOnly(CurrencyAmount) {
	let Formatter = new Intl.NumberFormat('en-US', {
  		style: 'currency',
  		currency: 'USD',
  		minimumFractionDigits: 0, // (this suffices for whole numbers, but will print 2500.10 as $2,500.1)
  		maximumFractionDigits: 0, // (causes 2500.99 to be printed as $2,501)
	});
	//console.log("FDO CurrencyAmount-"+ CurrencyAmount)
	return Formatter.format(CurrencyAmount)
}

export function SquareFootChange() {
	console.log("SquareFootChange  Start");
	$w("#PropertySqftInput").value = formatNumberComma(cleanstring($w("#PropertySqftInput").value))
	console.log("SquareFootChange Sqft-" + $w("#PropertySqftInput").value)

	if ($w("#PropertyPricePerSqFtInput").value !== "" && $w("#PropertySqftInput").value !== "") {
		$w("#PropertyDwellingAmountInput").value = formatDollarsOnly(Number(cleanstring($w("#PropertyPricePerSqFtInput").value)) * Number(cleanstring($w("#PropertySqftInput").value)));
		PolicyQuoteClear();
	}

	if ($w("#PropertyZipcodeInput").value !== '' && $w("#PropertySqftInput").value !== '') {

 		ReplacementCostCalculation(1, cleanstring($w("#PropertySqftInput").value), $w("#PropertyZipcodeInput").value).then(function(ParmReturn) {
			const ReplacementCostResult = ParmReturn;
			console.log("SFC LowReplacementCost-" + ReplacementCostResult.LowReplacementCost + ", HighReplacementCost-" + ReplacementCostResult.HighReplacementCost)
			$w("#LowReplacementCost").text = formatDollarsCents(ReplacementCostResult.LowReplacementCost);
			$w("#HighReplacementCost").text = formatDollarsCents(ReplacementCostResult.HighReplacementCost);

			}
		)
	}
	console.log("SquareFootChange  End");
}

export function SquareFootGet() {
	var City = "";
	var State = "";
	
	console.log("SquareFootGet  Start");
	if (($w('#PropertyStreetAddressInput').value !== "") && ($w('#PropertyZipcodeInput').value !== "")) {
		let StreetAddress = $w('#PropertyStreetAddressInput').value;
		let Zipcode = $w('#PropertyZipcodeInput').value;
		CityStateGet(Zipcode)
		.then(Result => {
			var CityStateReturn = Result;
			console.log("SquareFootGet - CityStateReturn-" + CityStateReturn.City);
			if (CityStateReturn.City == "") {
				$w("#ErrorMessageDisplay1").text = 'Error - Zipcode Not Found';
				$w("#ErrorMessageDisplay1").show()
			} else {
				City = CityStateReturn.City;
				State = CityStateReturn.State;
				
				console.log("SquareFootGet  City-" + City + ", State-" + State);

				FloodSquareFootGet(StreetAddress,City,State, Zipcode) 
				.then(Result => {
					FloodQuote = Result;
					console.log("SquareFootGet Flood Returned-" + JSON.stringify(FloodQuote.QuoteInfo));
					if (FloodQuote.QuoteInfo.errorList.length === 0) {
						console.log("SquareFootGet Flood SqFt-" + FloodQuote.QuoteInfo.application.squareFeet);
						$w('#PropertySqftInput').value = FloodQuote.QuoteInfo.application.squareFeet;
						SquareFootChange();

					} else {
						console.log("SquareFootGet Error Returned");
						console.log("SquareFootGet Error-" + FloodQuote.QuoteInfo.errorList[0].Error);
						$w("#ErrorMessageDisplay1").text = 'Flood Error' + " - " + FloodQuote.QuoteInfo.errorList[0].Error;
						$w("#ErrorMessageDisplay1").show()
					}
				})
				.catch (error => console.log('SquareFoodGet Call CError-' + error))
			}

		})
		.catch (error => console.log('CityStateGet Call CError-' + error))
	}
	console.log("SquareFootGet  End");
}
